def port_min_var(rets): # los pesos que hacen que el riesgo TOTAL sea lo más pequeño posible

    # matriz de covarianza de los rendimientos
    cov = rets.cov()
    # cuantos activos hay
    n = len(rets.columns)
    # varianza del portafolio / función objetivo
    var = lambda w: w.T @ cov @ w
    # pesos iniciales
    w0 = np.ones(n)/n
    # cotas
    bounds =[(0, 1)] * n
    # restricciones, tiene que sumar 1
    constraint =lambda w: sum(w)-1

    result = minimize(
        fun=var,
        x0=w0,
        bounds=bounds,
        constraints={'fun': constraint, 'type': 'eq'},
        tol=1e-20
    )

    return result.x

def max_sharpe(rets, rf):
    #rendimientos
    E=rets.mean()
    # matriz de covarianza
    cov=rets.cov()

    n_assets = len(rets.columns)

    r_sharpe = lambda w: -((w.T @ E - rf/252) / np.sqrt(w.T @ cov @ w)) # tasa se hace diaria si los redimientos son diarios
    #pesos iniciales
    w0=np.ones(n_assets)/n_assets
    #cotas
    bounds=[(0, 1)]*n_assets

    constraint=lambda w: sum(w)-1

    result = minimize(
        fun=r_sharpe,
        x0=w0,
        bounds=bounds,
        constraints={'fun': constraint, 'type': 'eq'},
        tol=1e-20
    )

    return result.x

def semi_var_target(returns: pd.DataFrame, returns_b: pd.DataFrame, corr: pd.DataFrame):

    # Alinea los datos de los rendimientos y el benchmark
    returns, returns_b = returns.align(returns_b, join='inner', axis=0)

    # Calcula la diferencia entre los rendimientos del portafolio y el benchmark
    diff_returns = returns - returns_b.values

    # rendimientos por debajo del benchmark
    downside_returns = diff_returns[diff_returns < 0].fillna(0)

    # Calcula el riesgo a la baja (desviación estándar de las pérdidas relativas)
    downside_risk = np.array(downside_returns.std())

    # Crea la matriz de semivarianza
    semi_var_matrix = corr * (downside_risk.reshape(len(returns.columns), 1) @ downside_risk.reshape(1, len(returns.columns))
    )

    # Define la función objetivo que es minimizar la semivarianza del portafolio
    target = lambda w: w.T @ semi_var_matrix @ w

    n = len(returns.keys())

    # Pesos iniciales iguales para todos los activos
    w_inicial = np.ones(n) / n

    # Límites de los pesos
    bounds = [(0, 1)] * n

    # Tolerancia
    tol = 1e-20

    # Restricción
    constraint = lambda w: np.sum(w) - 1

    # Optimización para minimizar la semivarianza
    trg = minimize(
        fun=target,
        x0=w_inicial,
        bounds=bounds,
        constraints={'fun': constraint, 'type': 'eq'},
        tol=tol
    )

    # pesos óptimos
    w_target = trg.x

    # Diccionario
    return dict(zip(returns.columns, w_target))

def max_omega(rets: pd.DataFrame):
    n = len(rets.columns)
    w_inicial = np.ones(n) / n
    bounds = [(0, 1)] * n
    tol = 1e-20

    rets_below_zero = rets[rets < 0].fillna(0)
    rets_above_zero = rets[rets > 0].fillna(0)
    upside_risk=rets_above_zero.std()
    downside_risk = rets_below_zero.std()

    omega_ind = upside_risk / downside_risk

    omega_neg = lambda w: -np.sum(w * omega_ind)

    constraints = {'fun': lambda w: np.sum(w) - 1, 'type': 'eq'}

    result = minimize(
        fun=omega_neg,
        x0=w_inicial,
        bounds=bounds,
        constraints=constraints,
        tol=tol
    )

    w_omega = result.x
    return dict(zip(rets.columns, w_omega))

def backtesting(prices: pd.DataFrame, benchmark: pd.DataFrame, weights: dict, capital: float):

    rets_assets = prices.pct_change().dropna()
    rets_benchmark = benchmark.pct_change().dropna()

    df = pd.DataFrame(index=rets_assets.index)

    # evolución
    for strategy in weights.keys():
        # para benchmark
        if strategy == 'benchmark':
            temp = rets_benchmark + 1
            temp.iloc[0]=capital
            temp = temp.cumprod()
            df[strategy] = temp.values

        else:
            temp = (rets_assets * weights.get(strategy)).sum(axis=1)+1
            temp.iloc[0]=capital
            temp = temp.cumprod()
            df[strategy] = temp.values

    # Gráfica
    plt.figure(figsize=(12,6))
    plt.plot(df, label=df.keys())
    plt.title('Backtesting of Strategies')
    plt.xlabel('Date')
    plt.ylabel('Portafolio Value')
    plt.legend()
    plt.show()


    return df

def metricas_desempeño(df, rf): # si tus datos los vas a anualizar, entonces el rf también anual !!!
    # Rendimientos
    rets= df.pct_change().dropna()

    # Matriz de Covarianza de los rendimientos
    cov_df = rets.cov()

    # Dataframe
    metricas_de_desempeño = pd.DataFrame(index=rets.columns)

    # Rendimiento
    metricas_de_desempeño['Rendimiento promedio anual'] = R_p = rets.mean()*252*100 # todo aquí esta en anual (cuidado)

    # Volatilidad
    metricas_de_desempeño['Volatilidad Anual'] = vol_a = rets.std()*np.sqrt(252)*100

    # Ratio de Sharpe
    metricas_de_desempeño['Ratio de Sharpe'] = (R_p - rf) / vol_a

    # Downside
    metricas_de_desempeño['Downside Risk'] = downside = rets[rets < 0].std() * np.sqrt(252) * 100

    # Upside
    metricas_de_desempeño['Upside Risk'] = upside = rets[rets > 0].std() * np.sqrt(252) * 100

    # Omega
    metricas_de_desempeño['Omega'] = omega = upside / downside

    # Beta
    covs = cov_df['Benchmark']
    varianza_market = covs['Benchmark']
    metricas_de_desempeño['Beta'] = beta = covs / varianza_market

    # Alfa de Jensen
    R_m = rets['Benchmark'].mean() * 252 * 100
    metricas_de_desempeño['Alpha']=alpha_Jensen = R_p - (rf + beta * (R_m - rf))

    # Ratio de Treynor
    metricas_de_desempeño['Ratio de Treynor'] = (R_p - rf) / beta

    # Ratio de Sortino
    metricas_de_desempeño['Ratio de Sortino'] = (R_p - rf) / downside

    return metricas_de_desempeño.T

Dentro de ellas, Max Omega destacó con un rendimiento de 18.7% y una volatilidad de 26.7%, logrando un equilibrio entre rendimiento-riesgo.

Max Omega presentó el mejor desempeño ajustado por riesgo total, sistemático y a la baja, respectivamente. Su Sharpe (0.70) y Sortino (1.14) fueron los más altos, indicando una capacidad superior para generar retornos consistentes frente a la variabilidad y las pérdidas. Min SemiVar mostró un perfil defensivo, con bajo riesgo a la baja y buena estabilidad, aunque con menor potencial de crecimiento. Las estrategias Min Var y Max Sharpe fueron las más conservadoras, sacrificando rentabilidad a cambio de menor exposición al riesgo.

Solo Max Omega obtuvo un Alpha positivo (10.7), evidenciando generación de valor por encima del rendimiento esperado según su riesgo.

En conjunto, la estrategia Max Omega se posiciona como la más eficiente y rentable, ofreciendo una relación riesgo-retorno superior y una mejor gestión del riesgo a la baja. Min SemiVar se distingue como la alternativa más estable y conservadora, mientras que el Benchmark representa una exposición total al mercado, con alto potencial de ganancia pero sin control del riesgo.
